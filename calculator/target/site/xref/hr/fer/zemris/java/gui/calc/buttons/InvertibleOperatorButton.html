3   /**
4    * InvertibleOperatorButton is a {@link OperatorButton} which has two states:
5    * normal and inverse. Inverse state has inverse binary operator of normal
6    * state. E.g. for operation x^n inverse operation would be x^(1/n).
7    * 
8    * @author Luka Skugor
9    *
10   */
11  public class InvertibleOperatorButton extends OperatorButton implements
12  		InvertibleButton {
13  
14  	/**
15  	 * 
16  	 */
17  	private static final long serialVersionUID = 1L;
18  	/**
19  	 * Display name when button is inverted.
20  	 */
21  	private String nameInverted;
22  	/**
23  	 * Binary operator of inverted state.
24  	 */
25  	private BinaryOperator invertedOperator;
26  
27  	/**
28  	 * Flag which indicates if button is inverted.
29  	 */
30  	boolean inverted;
31  
32  	/**
33  	 * Creates a new InvertibleOperatorButton with given names and binary
34  	 * operations for both states.
35  	 * 
36  	 * @param name display name of normal state
37  	 * @param nameInverted display name inverted state
38  	 * @param normalOperator binary operation of normal state
39  	 * @param invertedOperator binary operation of inverted state
40  	 */
41  	public InvertibleOperatorButton(String name, String nameInverted,
42  			BinaryOperator normalOperator, BinaryOperator invertedOperator) {
43  		super(name, normalOperator);
44  		this.nameInverted = nameInverted;
45  		this.invertedOperator = invertedOperator;
46  	}
47  
48  	/*
49  	 * (non-Javadoc)
50  	 * 
51  	 * @see hr.fer.zemris.java.gui.calc.buttons.OperatorButton#getOperator()
52  	 */
53  	@Override
54  	public BinaryOperator getOperator() {
55  		return (inverted) ? invertedOperator : operator;
56  	}
57  
58  	/*
59  	 * (non-Javadoc)
60  	 * 
61  	 * @see hr.fer.zemris.java.gui.calc.buttons.InvertibleButton#invert(boolean)
62  	 */
63  	public void invert(boolean inverted) {
64  		this.inverted = inverted;
65  		setText((inverted) ? nameInverted : name);
66  	}
67  
68  }
