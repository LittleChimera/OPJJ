14  	public static void main(String[] args) {
15  		
16  		if (args.length < 1) {
17  			System.err.println("Expected at least one action argument!");
18  			System.exit(1);
19  		}
20  		
21  		Map<String, ICryptoAction> actions = new HashMap<String, ICryptoAction>();
22  		actions.put("checksha", new CheckshaAction());
23  		actions.put("encrypt", new CryptAction(true));
24  		actions.put("decrypt", new CryptAction(false));
25  		
26  		String actionName = args[0];
27  		List<String> arguments = Arrays.asList(Arrays.copyOfRange(args, 1, args.length));
28  		
29  		try {
30  			actions.get(actionName).execute(arguments);
31  		} catch (NullPointerException e) {
32  			System.err.format("Action \"%s\" doesn't exists.%n", actionName);
33  			System.exit(2);
34  		} catch (RuntimeException e) {
35  			System.err.println(e.getLocalizedMessage());
36  			System.exit(3);
37  		}
38  	}
