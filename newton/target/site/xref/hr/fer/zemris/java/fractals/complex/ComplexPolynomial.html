1   package hr.fer.zemris.java.fractals.complex;
2   
3   import java.util.Arrays;
4   import java.util.Collections;
5   
6   /**
7    * ComplexPolynomial represents a polynomial which factors are {@link Complex}.
8    * Class instances are immutable.
9    * 
10   * @author Luka Skugor
11   *
12   */
13  public final class ComplexPolynomial {
14  
15  	/**
16  	 * Polynomial's factors from those who stand next to variable with greater
17  	 * power to the one with power zero.
18  	 */
19  	private Complex[] factors;
20  
21  	/**
22  	 * Creates a new polynomial with given factors. Factors are given from those
23  	 * which stand to variables with higher power. For example for polynomial
24  	 * 2*x^2 + 3*x + 1 given factors should be 2, 3, 1.
25  	 * 
26  	 * @param factors polynomial's factors
27  	 */
28  	public ComplexPolynomial(Complex... factors) {
29  		Collections.reverse(Arrays.asList(factors));
30  		factors = normalizeFactors(factors);
31  		if (factors.length == 0) {
32  			throw new IllegalArgumentException();
33  		}
34  		this.factors = factors;
35  	}
36  
37  	/**
38  	 * Returns order of this polynomial; e.g. For (7+2i)z^3+2z^2+5z+1 returns 3
39  	 * @return order of the polynomial
40  	 */
41  	public short order() {
42  		return (short) (factors.length - 1);
43  	}
44  
45  	/**
46  	 * Returns a new polynomial which is a result of this polynomial multiplied with given one.
47  	 * @param p polynomial multiplier
48  	 * @return resulting polynomial
49  	 */
50  	public ComplexPolynomial multiply(ComplexPolynomial p) {
51  		Complex[] factors = new Complex[this.order() + p.order() + 1];
52  		Arrays.fill(factors, Complex.ZERO);
53  		for (int i = 0; i < p.factors.length; i++) {
54  			for (int j = 0; j < this.factors.length; j++) {
55  				factors[i + j] = factors[i + j].add(p.factors[i]
56  						.multiply(this.factors[j]));
57  			}
58  		}
59  		Collections.reverse(Arrays.asList(factors));
60  		return new ComplexPolynomial(factors);
61  	}
62  
63  	/**
64  	 * Computes first derivative of this polynomial; for example, for (7+2i)z^3+2z^2+5z+1 returns (21+6i)z^2+4z+5
65  	 * @return derivative of this polynomial
66  	 */
67  	public ComplexPolynomial derive() {
68  		Complex[] derivedFactors = new Complex[factors.length - 1];
69  		for (int i = 1; i < factors.length; i++) {
70  			derivedFactors[i - 1] = factors[i].multiply(new Complex(i, 0));
71  		}
72  		Collections.reverse(Arrays.asList(derivedFactors));
73  		return new ComplexPolynomial(derivedFactors);
74  	}
75  
76  	/**
77  	 * Computes polynomial value at given point z.
78  	 * @param z complex point at which polynomial is calculated
79  	 * @return result for the applied point z
80  	 */
81  	public Complex apply(Complex z) {
82  		Complex result = new Complex(0, 0);
83  		for (int i = 0; i < factors.length; i++) {
84  
85  			Complex powedFactor = new Complex(1, 0);
86  			for (int j = 0; j < i; j++) {
87  				powedFactor = powedFactor.multiply(z);
88  			}
89  			result = result.add(factors[i].multiply(powedFactor));
90  		}
91  		return result;
92  	}
93  
94  	/*
95  	 * (non-Javadoc)
96  	 * 
97  	 * @see java.lang.Object#toString()
98  	 */
99  	@Override
100 	public String toString() {
101 		StringBuilder stringBuilder = new StringBuilder();
102 		for (int i = factors.length - 1; i >= 0; i--) {
103 			if (factors[i].equals(Complex.ZERO)) {
104 				continue;
105 			}
106 
107 			String rootString = factors[i].toString();
108 			if (rootString.trim().startsWith("-")) {
109 				rootString = factors[i].negate().toString();
110 				stringBuilder.append("-");
111 			} else if (stringBuilder.length() != 0) {
112 				stringBuilder.append("+");
113 			}
114 
115 			if (rootString.split("[\\+\\-]").length >= 2) {
116 				rootString = "(" + rootString + ")";
117 			}
118 			stringBuilder.append(rootString);
119 			if (i != 0) {
120 				stringBuilder.append("*z^").append(i);
121 			}
122 		}
123 		return stringBuilder.toString();
124 	}
125 
126 	/**
127 	 * Normalizes factors by removing first zeros, e.g. for factors 0, 0, 1, 0, 5 removes first 2 zeros.
128 	 * @param factors polynomial's factors
129 	 * @return normalized factors
130 	 */
131 	private static Complex[] normalizeFactors(Complex[] factors) {
132 		int order = factors.length;
133 		for (int i = factors.length - 1; i >= 0; i--) {
134 			if (!factors[i].equals(Complex.ZERO)) {
135 				break;
136 			}
137 			order--;
138 		}
139 
140 		return Arrays.copyOf(factors, order);
141 	}
